Example

local Rx = require "RxLua"

Rx.Observable.create(function (e)
    e:onNext("Hello")
    e:onNext("World")
    e:onComplete()
end)
:subscribe(
    print, 
    function (t)
        print("an error occured!", t)
    end, 
    function ()
        print("Completed!")
    end 
)
