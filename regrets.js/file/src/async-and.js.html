import { all } from './utils';
/**
 * Asynchronously apply a logical conjunction.
 * @param {?Promise} a
 * @param {?Promise} b
 * @return {Promise}
 */
const And = (a, b) => all([a, b]).then(v => v[0] && v[1]);
export default And;
